#### 2. تشغيل نماذج اللغة محلياً

##### تشغيل نموذج Gemma3
```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
import time

# تحميل النموذج والم-tokenizer
model_name = "google/gemma-3-27b-it"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(
    model_name,
    torch_dtype=torch.bfloat16,
    device_map="auto"
)

# دالة لتوليد النص
def generate_text(prompt, max_length=2048):
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    
    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_length=max_length,
            temperature=0.7,
            do_sample=True,
            top_p=0.95,
            top_k=40,
            pad_token_id=tokenizer.eos_token_id
        )
    
    response = tokenizer.decode(outputs[0], skip_special_tokens=True)
    return response[len(prompt):].strip()

# مثال الاستخدام
prompt = "ما هو الذكاء الاصطناعي؟"
response = generate_text(prompt)
print(response)
```

```python
from diffusers import StableDiffusionPipeline
import torch

# تحميل النموذج
model_id = "stabilityai/stable-diffusion-2-1"
pipe = StableDiffusionPipeline.from_pretrained(
    model_id,
    torch_dtype=torch.float16,
    variant="fp16"
).to("cuda")

# دالة لتوليد الصور
def generate_image(prompt, num_inference_steps=50, guidance_scale=7.5):
    image = pipe(
        prompt,
        num_inference_steps=num_inference_steps,
        guidance_scale=guidance_scale
    ).images[0]
    
    # حفظ الصورة
    image.save("generated_image.png")
    return image

# مثال الاستخدام
prompt = "صورة لغروب الشمس على شاطئ جميل مع أمواج هادئة"
```

##### تشغيل نموذج Kandinsky
```python
from diffusers import KandinskyPipeline, KandinskyPriorPipeline
import torch

# تحميل النماذج
prior_pipeline = KandinskyPriorPipeline.from_pretrained(
    "kandinsky-community/kandinsky-2-1-prior",
    torch_dtype=torch.float16
).to("cuda")

pipeline = KandinskyPipeline.from_pretrained(
    "kandinsky-community/kandinsky-2-1",
    torch_dtype=torch.float16
).to("cuda")

# دالة لتوليد الصور
def generate_image(prompt, num_inference_steps=50):
    # توليد embeddings الصورة
    image_embeds, negative_image_embeds = prior_pipeline(
        prompt,
        negative_prompt="low quality, bad quality",
        num_inference_steps=num_inference_steps,
        guidance_scale=4.0
    ).to_tuple()
    
    # توليد الصورة
    image = pipeline(
        image_embeds=image_embeds,
        negative_image_embeds=negative_image_embeds,
        num_inference_steps=num_inference_steps,
        height=768,
        width=768
    ).images[0]
    
    # حفظ الصورة
    image.save("generated_image.png")
    return image

# مثال الاستخدام
prompt = "صورة لغابة سحرية في الليل مع نجوم متلألئة"
image = generate_image(prompt)
print("تم حفظ الصورة باسم generated_image.png")
```

##### إعداد خادم بسيط باستخدام Flask
```python
from flask import Flask, request, jsonify
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
from diffusers import StableDiffusionPipeline
import base64
from io import BytesIO
from PIL import Image

app = Flask(__name__)

# تحميل النماذج
print("Loading models...")
tokenizer = AutoTokenizer.from_pretrained("google/gemma-3-27b-it")
text_model = AutoModelForCausalLM.from_pretrained(
    "google/gemma-3-27b-it",
    torch_dtype=torch.bfloat16,
    device_map="auto"
)

image_model = StableDiffusionPipeline.from_pretrained(
    "stabilityai/stable-diffusion-2-1",
    torch_dtype=torch.float16,
    variant="fp16"
).to("cuda")
print("Models loaded successfully!")

@app.route('/api/generate-text', methods=['POST'])
def generate_text():
    data = request.json
    prompt = data.get('prompt', '')
    
    if not prompt:
        return jsonify({'error': 'No prompt provided'}), 400
    
    inputs = tokenizer(prompt, return_tensors="pt").to(text_model.device)
    
    with torch.no_grad():
        outputs = text_model.generate(
            **inputs,
            max_length=2048,
            temperature=0.7,
            do_sample=True,
            top_p=0.95,
            top_k=40,
            pad_token_id=tokenizer.eos_token_id
        )
    
    response = tokenizer.decode(outputs[0], skip_special_tokens=True)
    response = response[len(prompt):].strip()
    
    return jsonify({
        'response': response,
        'prompt': prompt
    })

@app.route('/api/generate-image', methods=['POST'])
def generate_image():
    data = request.json
    prompt = data.get('prompt', '')
    
    if not prompt:
        return jsonify({'error': 'No prompt provided'}), 400
    
    image = image_model(
        prompt,
        num_inference_steps=50,
        guidance_scale=7.5
    ).images[0]
    
    # تحويل الصورة إلى base64
    buffered = BytesIO()
    image.save(buffered, format="PNG")
    img_str = base64.b64encode(buffered.getvalue()).decode('utf-8')
    
    return jsonify({
        'image': img_str,
        'prompt': prompt
    })

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
```

##### إعداد خادم متقدم باستخدام FastAPI
```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Optional
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
from diffusers import StableDiffusionPipeline
import base64
from io import BytesIO
from PIL import Image
import asyncio
from concurrent.futures import ThreadPoolExecutor

app = FastAPI(title="QUSAY Pro Local API")

# نماذج Pydantic للطلبات والاستجابات
class TextRequest(BaseModel):
    prompt: str
    max_length: Optional[int] = 2048
    temperature: Optional[float] = 0.7

class TextResponse(BaseModel):
    response: str
    prompt: str

class ImageRequest(BaseModel):
    prompt: str
    num_inference_steps: Optional[int] = 50
    guidance_scale: Optional[float] = 7.5

class ImageResponse(BaseModel):
    image: str
    prompt: str

# تحميل النماذج
print("Loading models...")
tokenizer = AutoTokenizer.from_pretrained("google/gemma-3-27b-it")
text_model = AutoModelForCausalLM.from_pretrained(
    "google/gemma-3-27b-it",
    torch_dtype=torch.bfloat16,
    device_map="auto"
)

image_model = StableDiffusionPipeline.from_pretrained(
    "stabilityai/stable-diffusion-2-1",
    torch_dtype=torch.float16,
    variant="fp16"
).to("cuda")

# منفذ تنفيذ المهام
executor = ThreadPoolExecutor(max_workers=4)

@app.post("/api/generate-text", response_model=TextResponse)
async def generate_text(request: TextRequest):
    try:
        # تشغيل توليد النص في منفذ منفصل
        loop = asyncio.get_event_loop()
        response = await loop.run_in_executor(
            executor,
            _generate_text_sync,
            request.prompt,
            request.max_length,
            request.temperature
        )
        
        return TextResponse(response=response, prompt=request.prompt)
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

def _generate_text_sync(prompt, max_length, temperature):
    inputs = tokenizer(prompt, return_tensors="pt").to(text_model.device)
    
    with torch.no_grad():
        outputs = text_model.generate(
            **inputs,
            max_length=max_length,
            temperature=temperature,
            do_sample=True,
            top_p=0.95,
            top_k=40,
            pad_token_id=tokenizer.eos_token_id
        )
    
    response = tokenizer.decode(outputs[0], skip_special_tokens=True)
    return response[len(prompt):].strip()

@app.post("/api/generate-image", response_model=ImageResponse)
async def generate_image(request: ImageRequest):
    try:
        # تشغيل توليد الصورة في منفذ منفصل
        loop = asyncio.get_event_loop()
        image_b64 = await loop.run_in_executor(
            executor,
            _generate_image_sync,
            request.prompt,
            request.num_inference_steps,
            request.guidance_scale
        )
        
        return ImageResponse(image=image_b64, prompt=request.prompt)
```

```javascript
function getBrowserFingerprint() {
    // إنشاء بصمة متصفح فريدة للمستخدم
    const fingerprint = [
        navigator.userAgent,
        navigator.language,
        screen.colorDepth,
        screen.width,
        screen.height,
        new Date().getTimezoneOffset(),
        !!window.sessionStorage,
        !!window.localStorage,
        !!window.indexedDB,
        navigator.hardwareConcurrency,
        navigator.deviceMemory || 4
    ].join('|');
    
    // إنشاء تجزئة بسيطة للبصمة
    let hash = 0;
    for (let i = 0; i < fingerprint.length; i++) {
        const char = fingerprint.charCodeAt(i);
        hash = ((hash << 5) - hash) + char;
        hash = hash & hash; // تحويل إلى عدد صحيح 32 بت
    }
    
    return Math.abs(hash).toString(16);
}
```

```javascript
class RateLimiter {
    constructor() {
        this.limits = new Map();
        this.defaultLimits = {
            imageGeneration: 5000, // 5 ثواني بين كل طلب توليد صورة
            chatRequest: 3000,     // 3 ثواني بين كل طلب دردشة
            webSearch: 2000,       // 2 ثانية بين كل طلب بحث
            weatherRequest: 60000, // دقيقة واحدة بين كل طلب طقس
            qrGeneration: 1000    // ثانية واحدة بين كل طلب توليد QR
        };
        
        // استخراج IP المستخدم
        this.userIP = this.getBrowserFingerprint();
        this.initializeUserLimits();
    }
    
    canMakeRequest(type) {
        const userLimits = this.limits.get(this.userIP);
        if (!userLimits) {
            this.initializeUserLimits();
            return this.canMakeRequest(type);
        }
        
        const now = Date.now();
        const lastRequest = userLimits[`last${type.charAt(0).toUpperCase() + type.slice(1)}`] || 0;
        const limit = this.defaultLimits[type] || 1000;
        
        return (now - lastRequest) >= limit;
    }
    
    recordRequest(type) {
        const userLimits = this.limits.get(this.userIP);
        if (!userLimits) {
            this.initializeUserLimits();
            return this.recordRequest(type);
        }
        
        userLimits[`last${type.charAt(0).toUpperCase() + type.slice(1)}`] = Date.now();
    }
    
    getTimeUntilNextRequest(type) {
        const userLimits = this.limits.get(this.userIP);
        if (!userLimits) {
            return 0;
        }
        
        const now = Date.now();
        const lastRequest = userLimits[`last${type.charAt(0).toUpperCase() + type.slice(1)}`] || 0;
        const limit = this.defaultLimits[type] || 1000;
        const timePassed = now - lastRequest;
        
        return Math.max(0, limit - timePassed);
    }
}
```

- **التخزين**: 2GB مساحة حرة
- **الشبكة**: اتصال إنترنت مستقر

### خطوات التركيب

#### 1. استنساخ المستودع
```bash
# استنساخ المستودع
git clone https://github.com/username/QUSAY-Pro.git

# الدخول إلى مجلد المشروع
cd QUSAY-Pro
```

#### 2. تثبيت الاعتماديات
```bash
# تثبيت اعتماديات Node.js
npm install

# تثبيت اعتماديات Python (إذا لزم الأمر)
pip install -r requirements.txt
```

#### 3. إعداد المتغيرات البيئية
```bash
# نسخ ملف المتغيرات البيئية
cp .env.example .env

# تعديل المتغيرات البيئية
nano .env
```

#### 4. تشغيل المشروع
```bash
# تشغيل خادم التطوير
npm start

# أو
npm run dev
```

#### 5. الوصول إلى التطبيق
افتح المتصفح وانتقل إلى:
```
http://localhost:3000
```
